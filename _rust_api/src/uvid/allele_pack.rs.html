1//! Two-bit nucleotide encoding using conventional bioinformatics ordering.
2//!
3//! Encoding: A=0b00, C=0b01, G=0b10, T=0b11
4//!
5//! This gives a clean 4-element lookup for decode and a simple match for encode.
6//!
7//! ## Per-allele 46-bit encoding
8//!
9//! Each allele (REF or ALT) is encoded into 46 bits using a mode bit:
10//!
11//! ```text
12//! Mode 0 (string mode, bit 45 = 0): exact 2-bit encoded sequence
13//!   Bit 45:      0 (string mode)
14//!   Bits 44-40:  5-bit sequence length (0-20)
15//!   Bits 39-0:   up to 20 bases of 2-bit DNA (40 bits), zero-padded right
16//!
17//! Mode 1 (length mode, bit 45 = 1): length + Rabin fingerprint
18//!   Bit 45:      1 (length mode)
19//!   Bits 44-17:  28-bit integer length (max 268,435,455 — covers chr1 at 248M)
20//!   Bits 16-0:   17-bit Rabin fingerprint of the sequence
21//! ```
22//!
23//! String mode sorts before length mode (0 < 1) at the same locus.
24//! In length mode, the actual sequence is not stored; decode returns N-repeats.
25//!
26//! ## Rabin fingerprint for sequence discrimination
27//!
28//! Length-mode alleles store a 17-bit Rabin fingerprint of the sequence to
29//! reduce collisions between variants at the same locus with the same allele
30//! length but different sequences (e.g. insertions of nearly-identical tandem
31//! repeats differing by a few SNPs).
32//!
33//! The fingerprint uses polynomial division in GF(2) with an irreducible
34//! polynomial of degree 17.  The DNA sequence is treated as a bitstream where
35//! each base contributes 2 bits (A=00, C=01, G=10, T=11), shifted into the
36//! register MSB-first.  Non-ACGT characters (e.g. N) are treated as A (0b00),
37//! which is the most common base; this is acceptable because N-containing
38//! sequences in length mode are extremely rare and the fingerprint is a
39//! best-effort discriminator, not a guarantee.
40//!
41//! **Irreducible polynomial**: `x^17 + x^3 + 1` (binary: `0x20009`).
42//! This is a known irreducible polynomial over GF(2) chosen for its sparsity
43//! (only 3 non-zero terms), which minimises the cost of the reduction step.
44//!
45//! **Properties**:
46//! - Position-sensitive: "ACG" and "GCA" produce different fingerprints
47//!   because the polynomial shift encodes positional information.
48//! - Single-base perturbation sensitive: changing one base anywhere in the
49//!   sequence changes the remainder of the polynomial division, producing
50//!   a different fingerprint with probability ~(1 - 2^-17) ≈ 99.999%.
51//! - Deterministic and cheap: only bitwise shifts and XORs, no multiplication.
52//! - 1-in-131,072 false collision rate for random same-length sequences.
53
54/// Encode a single nucleotide character to its 2-bit representation.
55///
56/// Returns `None` for non-ACGT characters.
57#[inline]
58pub fn encode_base(base: u8) -> Option<u8> {
59    match base {
60        b'A' | b'a' => Some(0b00),
61        b'C' | b'c' => Some(0b01),
62        b'G' | b'g' => Some(0b10),
63        b'T' | b't' => Some(0b11),
64        _ => None,
65    }
66}

68/// Decode a 2-bit value back to a nucleotide ASCII byte.
69///
70/// Only the lowest 2 bits of `bits` are considered.
71#[inline]
72pub fn decode_base(bits: u8) -> u8 {
73    const LOOKUP: [u8; 4] = [b'A', b'C', b'G', b'T'];
74    LOOKUP[(bits & 0b11) as usize]
75}

77/// Encode a DNA sequence into a u128, packed from the MSB side.
78///
79/// Returns the packed bits and the number of bases encoded.
80/// Bases are packed left-to-right starting at the most significant position
81/// within the allocated bit range.
82///
83/// `max_bases` limits how many bases will be encoded (for fitting into a bit budget).
84/// Returns `None` if any character is not a valid nucleotide.
85pub fn encode_sequence(seq: &[u8], max_bases: usize) -> Option<(u128, usize)> {
86    let count = seq.len().min(max_bases);
87    let mut packed: u128 = 0;
88    for (i, &base) in seq[..count].iter().enumerate() {
89        let bits = encode_base(base)? as u128;
90        // Pack from the left: first base goes into the highest bit positions
91        // within a 76-bit payload, the caller is responsible for shifting into
92        // the correct position within the full 128-bit word.
93        let shift = (count - 1 - i) * 2;
94        packed |= bits << shift;
95    }
96    Some((packed, count))
97}

99/// Decode a 2-bit packed sequence back into a byte vector of nucleotides.
100///
101/// `packed` contains the bases packed from MSB side.
102/// `count` is the number of bases to decode.
103pub fn decode_sequence(packed: u128, count: usize) -> Vec<u8> {
104    let mut seq = Vec::with_capacity(count);
105    for i in 0..count {
106        let shift = (count - 1 - i) * 2;
107        let bits = ((packed >> shift) & 0b11) as u8;
108        seq.push(decode_base(bits));
109    }
110    seq
111}

113/// Check if a byte slice is a valid DNA sequence (only A, C, G, T characters).
114pub fn is_valid_dna(seq: &[u8]) -> bool {
115    seq.iter()
116        .all(|&b| matches!(b, b'A' | b'a' | b'C' | b'c' | b'G' | b'g' | b'T' | b't'))
117}

121/// Number of bits in each allele field.
122pub const ALLELE_BITS: u32 = 46;
123
124/// Maximum number of bases that fit in string mode (40 payload bits / 2 bits per base).
125pub const MAX_STRING_BASES: usize = 20;
126
127/// Mask for the full 46-bit allele field.
128pub const ALLELE_MASK: u64 = (1u64 << ALLELE_BITS) - 1;
129
130/// The mode bit position within a 46-bit allele field.
131///
132/// Bit 45 = mode: 0 for string, 1 for length.
133/// Public so that `uvid128` can split allele values into mode + payload.
134pub const MODE_BIT_POS: u32 = 45;
135
136/// Mask for the mode bit.
137const MODE_MASK: u64 = 1u64 << MODE_BIT_POS;
138
139/// The mode bit position (internal alias).
140const MODE_BIT: u32 = MODE_BIT_POS;
141
142/// Mask for bits below the mode bit (the 45-bit payload).
143///
144/// Public so that `uvid128` can extract the payload from an allele value.
145pub const PAYLOAD_MASK: u64 = MODE_MASK - 1; // bits 44-0
146
147/// Number of bits for the sequence length prefix in string mode.
148const STRING_LEN_BITS: u32 = 5;
149
150/// Shift for the string length field (below mode bit, above sequence payload).
151const STRING_LEN_SHIFT: u32 = MODE_BIT - STRING_LEN_BITS; // 40
152
153/// Mask for the string length field (5 bits).
154const STRING_LEN_MASK: u64 = ((1u64 << STRING_LEN_BITS) - 1) << STRING_LEN_SHIFT;
155
156// (STRING_PAYLOAD_BITS removed — value equals STRING_LEN_SHIFT and was unused)
157
158// ── Length mode sub-layout constants ───────────────────────────────────────
159//
160// Length mode (bit 45 = 1): bits 44-0 split into 28-bit length + 17-bit Rabin hash.
161//
162// ```text
163// Bit 45:      1 (length mode)
164// Bits 44-17:  28-bit integer length (max 268,435,455)
165// Bits 16-0:   17-bit Rabin fingerprint
166// ```
167
168/// Number of bits for the Rabin fingerprint in length mode.
169pub const FINGERPRINT_BITS: u32 = 17;
170
171/// Number of bits for the integer length in length mode.
172pub const LENGTH_BITS: u32 = 28;
173
174/// Shift for the length field within the 45-bit payload (above the fingerprint).
175const LENGTH_SHIFT: u32 = FINGERPRINT_BITS; // 17
176
177/// Mask for the 17-bit fingerprint (bits 16-0 of the payload).
178const FINGERPRINT_MASK: u64 = (1u64 << FINGERPRINT_BITS) - 1;
179
180/// Mask for the 28-bit length field (bits 44-17 of the payload).
181const LENGTH_FIELD_MASK: u64 = ((1u64 << LENGTH_BITS) - 1) << LENGTH_SHIFT;
182
183/// Maximum length encodable in length mode (2^28 - 1 = 268,435,455).
184pub const MAX_LENGTH_MODE_LEN: usize = (1usize << LENGTH_BITS) - 1;
185
186/// Irreducible polynomial for Rabin fingerprint: x^17 + x^3 + 1 (binary).
187///
188/// Only the lower 17 bits are used for XOR reduction (the leading x^17 term
189/// is implicit in the shift-and-reduce algorithm).
190const RABIN_POLY: u32 = 0x0_0009; // bits: x^3 + x^0 = 0b0000_0000_0000_1001
191
192/// Result of decoding a 46-bit allele field.
193#[derive(Debug, Clone, PartialEq, Eq)]
194pub enum AlleleDecoded {
195    /// String mode: the exact DNA sequence was stored.
196    Sequence(Vec<u8>),
197    /// Length mode: the integer length and Rabin fingerprint were stored.
198    /// The actual bases are unknown (decode as N-repeats).
199    Length { len: usize, fingerprint: u32 },
200}
201
202/// Compute a 17-bit Rabin fingerprint of a DNA sequence.
203///
204/// The sequence is treated as a bitstream where each base contributes 2 bits
205/// (A=00, C=01, G=10, T=11), shifted into a polynomial division register
206/// MSB-first. Non-ACGT characters (including N) are treated as A (0b00).
207///
208/// Uses irreducible polynomial x^17 + x^3 + 1 over GF(2).
209///
210/// The result is in the range `[0, 2^17)` (0 to 131,071).
211pub fn rabin_fingerprint_dna(seq: &[u8]) -> u32 {
212    let mut hash: u32 = 0;
213
214    for &base in seq {
215        let bits: u32 = match base {
216            b'A' | b'a' => 0b00,
217            b'C' | b'c' => 0b01,
218            b'G' | b'g' => 0b10,
219            b'T' | b't' => 0b11,
220            _ => 0b00, // N and other non-ACGT → A
221        };
222
223        // Shift in 2 bits at a time (MSB first within the 2-bit pair).
224        // We need to check for overflow after each single-bit shift.
225
226        // Shift in high bit of the 2-bit pair
227        hash <<= 1;
228        hash |= (bits >> 1) & 1;
229        if hash & (1 << FINGERPRINT_BITS) != 0 {
230            hash ^= RABIN_POLY;
231            hash &= FINGERPRINT_MASK as u32;
232        }
233
234        // Shift in low bit of the 2-bit pair
235        hash <<= 1;
236        hash |= bits & 1;
237        if hash & (1 << FINGERPRINT_BITS) != 0 {
238            hash ^= RABIN_POLY;
239            hash &= FINGERPRINT_MASK as u32;
240        }
241    }
242
243    hash
244}

246/// Encode a single allele (REF or ALT) into a 46-bit packed value.
247///
248/// - Empty or `"."` → string mode with length 0 (represents deletion/missing).
249/// - Length 1-20 with valid ACGT bases → string mode with exact 2-bit sequence.
250/// - Length > 20 OR contains non-ACGT bases → length mode with integer length.
251///
252/// This function always succeeds (no `None` return).
253pub fn encode_allele(seq: &[u8]) -> u64 {
254    // Handle deletion symbol or empty
255    if seq.is_empty() || seq == b"." {
256        // String mode, length 0, no payload
257        return 0;
258    }
259
260    let len = seq.len();
261
262    // Try string mode: must be <= 20 bases and all valid ACGT
263    if len <= MAX_STRING_BASES && is_valid_dna(seq) {
264        // String mode: bit 45 = 0
265        let mut packed: u64 = 0;
266
267        // Pack length into bits 44-40
268        packed |= (len as u64) << STRING_LEN_SHIFT;
269
270        // Pack 2-bit encoded bases into bits 39-0 (from MSB side)
271        for (i, &base) in seq.iter().enumerate() {
272            let bits = encode_base(base).unwrap() as u64;
273            // First base at highest position (bits 39-38), etc.
274            let shift = (MAX_STRING_BASES - 1 - i) * 2;
275            packed |= bits << shift;
276        }
277
278        packed
279    } else {
280        // Length mode: bit 45 = 1, bits 44-17 = 28-bit length, bits 16-0 = 17-bit Rabin hash
281        let fingerprint = rabin_fingerprint_dna(seq) as u64;
282        MODE_MASK | ((len as u64) << LENGTH_SHIFT) | (fingerprint & FINGERPRINT_MASK)
283    }
284}

286/// Decode a 46-bit allele field back to its components.
287///
288/// Returns `AlleleDecoded::Sequence` for string mode (exact bases) or
289/// `AlleleDecoded::Length` for length mode (integer length only).
290pub fn decode_allele(packed: u64) -> AlleleDecoded {
291    if packed & MODE_MASK != 0 {
292        // Length mode: bits 44-17 are the 28-bit length, bits 16-0 are the 17-bit fingerprint
293        let payload = packed & PAYLOAD_MASK;
294        let len = ((payload & LENGTH_FIELD_MASK) >> LENGTH_SHIFT) as usize;
295        let fingerprint = (payload & FINGERPRINT_MASK) as u32;
296        AlleleDecoded::Length { len, fingerprint }
297    } else {
298        // String mode: bits 44-40 are the length, bits 39-0 are 2-bit DNA
299        let len = ((packed & STRING_LEN_MASK) >> STRING_LEN_SHIFT) as usize;
300        if len == 0 {
301            return AlleleDecoded::Sequence(Vec::new());
302        }
303        let mut seq = Vec::with_capacity(len);
304        for i in 0..len {
305            let shift = (MAX_STRING_BASES - 1 - i) * 2;
306            let bits = ((packed >> shift) & 0b11) as u8;
307            seq.push(decode_base(bits));
308        }
309        AlleleDecoded::Sequence(seq)
310    }
311}

313/// Check if a 46-bit allele field is in string mode (exact sequence stored).
314#[inline]
315pub fn is_string_mode(packed: u64) -> bool {
316    packed & MODE_MASK == 0
317}

612            b"ACGTACGTACGTACGTACGTA", // length mode, 21
613        ];
614        for case in cases {
615            let packed = encode_allele(case);
616            assert_eq!(
617                packed & !ALLELE_MASK,
618                0,
619                "Exceeds 46 bits for {:?}",
620                std::str::from_utf8(case)
621            );
622        }
623        // Max length mode value: mode=1, length=MAX_LENGTH_MODE_LEN, fingerprint=all 1s
624        let packed = MODE_MASK | ((MAX_LENGTH_MODE_LEN as u64) << LENGTH_SHIFT) | FINGERPRINT_MASK;
625        assert_eq!(packed & !ALLELE_MASK, 0, "Max length mode exceeds 46 bits");
626    }
