1//! Variant normalization algorithm
2//! ([Tan et al. 2015](https://doi.org/10.1093/bioinformatics/btv112)).
3//!
4//! Implements the "unified representation" normalization:
5//! 1. **Right-trim**: strip matching trailing bases from REF and ALT
6//! 2. **Left-trim**: strip matching leading bases, adjusting POS
7//! 3. **Left-align**: while last bases match, prepend upstream reference
8//!    base and shift POS left
9//!
10//! Steps 1-3 repeat until convergence (no changes in an iteration).
11//!
12//! This module is intentionally free of UVID-specific types.
13
14use super::error::NormalizeError;
15use super::reference::ReferenceGenome;
16use super::types::NormalizedVariant;
17
18/// Maximum iterations before we declare non-convergence.
19/// In practice, normalization converges in 1-2 iterations.
20const MAX_ITERATIONS: u32 = 100;
21
22/// Returns `true` if the allele is symbolic (e.g. `<DEL>`, `<INS>`)
23/// or a breakend (`]` / `[` notation), or missing (`.`), or the
24/// spanning deletion (`*`).
25fn is_symbolic_or_special(allele: &[u8]) -> bool {
26    if allele.is_empty() {
27        return true;
28    }
29    // Missing allele
30    if allele == b"." {
31        return true;
32    }
33    // Star (spanning deletion)
34    if allele == b"*" {
35        return true;
36    }
37    // Symbolic: <...>
38    if allele.first() == Some(&b'<') {
39        return true;
40    }
41    // Breakend notation
42    if allele.contains(&b'[') || allele.contains(&b']') {
43        return true;
44    }
45    false
46}
47
48/// Normalize a single variant using the
49/// [Tan et al. 2015](https://doi.org/10.1093/bioinformatics/btv112) algorithm.
50///
51/// # Arguments
52///
53/// * `chrom` — Chromosome/contig name (e.g. `"chr1"`, `"20"`)
54/// * `pos` — 1-based VCF position
55/// * `ref_seq` — Reference allele bytes (uppercase ASCII DNA)
56/// * `alt_seq` — Alternate allele bytes (uppercase ASCII DNA)
57/// * `reference` — Reference genome for fetching upstream bases
58///
59/// # Returns
60///
61/// A [`NormalizedVariant`] with the (possibly adjusted) position and
62/// allele sequences.  If the variant was already normalized or was
63/// skipped (SNV, symbolic), `was_modified` is `false`.
64///
65/// # Skipped variant types
66///
67/// - Symbolic alleles (`<DEL>`, `<INS>`, etc.)
68/// - Missing alleles (`.`)
69/// - Spanning deletions (`*`)
70/// - Breakend notation
71pub fn normalize(
72    chrom: &str,
73    pos: u32,
74    ref_seq: &[u8],
75    alt_seq: &[u8],
76    reference: &mut dyn ReferenceGenome,
77) -> Result<NormalizedVariant, NormalizeError> {
78    // Skip symbolic/special alleles
79    if is_symbolic_or_special(ref_seq) || is_symbolic_or_special(alt_seq) {
80        return Ok(NormalizedVariant::unchanged(chrom, pos, ref_seq, alt_seq));
81    }
82
83    // Skip if either allele is empty (shouldn't happen in valid VCF, but be safe)
84    if ref_seq.is_empty() || alt_seq.is_empty() {
85        return Ok(NormalizedVariant::unchanged(chrom, pos, ref_seq, alt_seq));
86    }
87
88    // Work with mutable copies
89    let mut r: Vec<u8> = ref_seq.to_vec();
90    let mut a: Vec<u8> = alt_seq.to_vec();
91    let mut p = pos;
92    let orig_pos = pos;
93
94    for _iter in 0..MAX_ITERATIONS {
95        let (r2, a2, p2) = normalize_one_pass(chrom, p, &r, &a, reference)?;
96
97        if r2 == r && a2 == a && p2 == p {
98            // Converged
99            let was_modified = p != orig_pos || r.as_slice() != ref_seq || a.as_slice() != alt_seq;
100            return Ok(NormalizedVariant {
101                chrom: chrom.to_string(),
102                pos: p,
103                ref_seq: r,
104                alt_seq: a,
105                was_modified,
106            });
107        }
108
109        r = r2;
110        a = a2;
111        p = p2;
112    }
113
114    Err(NormalizeError::DidNotConverge {
115        chrom: chrom.to_string(),
116        pos: orig_pos,
117        iterations: MAX_ITERATIONS,
118    })
119}

121/// One pass of the normalization algorithm: right-trim, left-trim, left-align.
122fn normalize_one_pass(
123    chrom: &str,
124    mut pos: u32,
125    ref_seq: &[u8],
126    alt_seq: &[u8],
127    reference: &mut dyn ReferenceGenome,
128) -> Result<(Vec<u8>, Vec<u8>, u32), NormalizeError> {
129    let mut r = ref_seq.to_vec();
130    let mut a = alt_seq.to_vec();
131
132    // --- Step 1: Right-trim ---
133    // Remove matching bases from the end, keeping at least 1 base each.
134    while r.len() > 1 && a.len() > 1 && r.last() == a.last() {
135        r.pop();
136        a.pop();
137    }
138
139    // --- Step 2: Left-trim ---
140    // Remove matching bases from the start, adjusting POS.
141    // Keep at least 1 base each.
142    while r.len() > 1 && a.len() > 1 && r.first() == a.first() {
143        r.remove(0);
144        a.remove(0);
145        pos += 1;
146    }
147
148    // --- Step 3: Left-align ---
149    // Only for indels: one allele is a prefix of the other, or after
150    // trimming we have one allele of length 1.
151    // While the last bases of REF and ALT match, prepend an upstream
152    // reference base and shift POS left.
153    //
154    // Skip left-alignment for complex/MNV variants (both > 1 base
155    // after trimming, and neither is a pure prefix of the other).
156    let is_indel = r.len() == 1 || a.len() == 1;
157
158    if is_indel {
159        while pos > 1 && r.last() == a.last() {
160            // Shift left: prepend the base at pos-2 (0-based: pos-1 is current,
161            // so upstream is pos-2)
162            let upstream_pos = pos - 1; // 1-based position of the base to prepend
163            let upstream_base = reference.get_sequence(chrom, upstream_pos - 1, upstream_pos)?; // 0-based [pos-2, pos-1)
164
165            if upstream_base.is_empty() {
166                break;
167            }
168
169            let base = upstream_base[0];
170
171            // Remove last base (they match)
172            r.pop();
173            a.pop();
174
175            // Prepend upstream base
176            r.insert(0, base);
177            a.insert(0, base);
178
179            pos -= 1;
180        }
181    }
182
183    Ok((r, a, pos))
184}

265    fn run_test_cases(cases: &[NormTestCase]) {
266        let mut reference = bcftools_norm_reference();
267        for tc in cases {
268            let result = normalize(tc.chrom, tc.pos, tc.ref_seq, tc.alt_seq, &mut reference)
269                .unwrap_or_else(|e| panic!("{}: normalization failed: {}", tc.name, e));
270            assert_eq!(
271                result.pos, tc.expected_pos,
272                "{} ({}): pos mismatch: got {}, expected {}",
273                tc.name, tc.source, result.pos, tc.expected_pos
274            );
275            assert_eq!(
276                result.ref_seq,
277                tc.expected_ref,
278                "{} ({}): ref mismatch: got {:?}, expected {:?}",
279                tc.name,
280                tc.source,
281                String::from_utf8_lossy(&result.ref_seq),
282                String::from_utf8_lossy(tc.expected_ref)
283            );
284            assert_eq!(
285                result.alt_seq,
286                tc.expected_alt,
287                "{} ({}): alt mismatch: got {:?}, expected {:?}",
288                tc.name,
289                tc.source,
290                String::from_utf8_lossy(&result.alt_seq),
291                String::from_utf8_lossy(tc.expected_alt)
292            );
293        }
294    }

691    // Poly-T region on chr4: deletion with left-alignment
692    // ------------------------------------------------------------------
693
694    #[test]
695    fn test_poly_t_deletion_chr4() {
696        // 4:21 ATTTTTTTTTTTTTTTC -> ATTTTTTTTTTTTTTC (1 T deletion)
697        // Right-trim C => ATTTTTTTTTTTTTTT / ATTTTTTTTTTTTTT
698        // Left-trim A => pos 22, TTTTTTTTTTTTTTT / TTTTTTTTTTTTTT
699        // Left-trim more T's => pos 36, TT / T
700        // Left-align: indel (ALT len 1), last bases T==T, prepend upstream
701        //   pos 35: ref[34]=T => TT/T... still match => continue
702        //   ... all the way back through poly-T run
703        // Eventually: pos 22, TT -> T (left edge of poly-T)
704        //
705        // Actually let's be more careful. After right-trim C:
706        // REF = ATTTTTTTTTTTTTTT (16 chars), ALT = ATTTTTTTTTTTTTT (15 chars)
707        // Left-trim: A matches => pos 22, REF=TTTTTTTTTTTTTTT (15 T's), ALT=TTTTTTTTTTTTTT (14 T's)
708        // Left-trim more T's (while both > 1): trim 13 more T's => pos 35, TT / T
709        // Left-align: indel. Last bases T == T. Prepend upstream:
710        //   pos 34 (0-based 33): ref="T" => pos 34, REF=TT, ALT=T. Still T==T.
711        //   ... continues to pos 22. ref[21] (0-based)="T"
712        //   pos 22: REF=TT, ALT=T. Last bases T==T.
713        //   pos 21: ref[20] (0-based)="C" => pos 21, REF=CT, ALT=C. C!=T, stop.
714        //   Wait, that gives pos 21, CT/C. But there's left-trim: C==C, trim => pos 22, T/"".
715        //   Empty ALT isn't valid, so we need the anchor base. Let me re-think.
716        //
717        // Actually the loop converges across iterations. After first pass:
718        // left-align gives us some position with CT/C or similar.
719        // Second pass: right-trim nothing, left-trim C => pos+1, T/empty.
720        // Empty isn't allowed... the algorithm keeps 1 base minimum.
721        //
722        // Let me just test what bcftools expects:
723        // bcftools output for 4:21 shows it gets decomposed into:
724        //   4:25 T/TT,* and 4:36 TC/C,TT,TTC
725        // But that's multiallelic split, not what we do per-allele.
726        //
727        // For the single deletion allele ATTTTTTTTTTTTTTTC -> ATTTTTTTTTTTTTTC:
728        // This is deletion of 1 T in a poly-T run. Should left-align to the
729        // leftmost position.
730        let mut reference = bcftools_norm_reference();
731
732        // Single allele: delete 1 T
733        let r = normalize(
734            "4",
735            21,
736            b"ATTTTTTTTTTTTTTTC",
737            b"ATTTTTTTTTTTTTTC",
738            &mut reference,
739        )
740        .unwrap();
741        // Should left-align through the poly-T run
742        // chr4 sequence: TCCCCTCTTGACCTCTCTCTATTTTTTTTTTTTTTTCTGAG...
743        //                 1234567890123456789012345678901234567890
744        // pos 21 (1-based) = 'A' (0-based index 20)
745        // The T-run starts at pos 22 (0-based 21)
746        // Left-align should push to pos 21: AT -> A (anchor A before T-run)
747        assert_eq!(r.pos, 21, "poly-T del pos");
748        assert_eq!(r.ref_seq, b"AT", "poly-T del ref");
749        assert_eq!(r.alt_seq, b"A", "poly-T del alt");
750    }

836        // Wait, let's be precise:
837        // REF = TAAACCCTAAA, ALT = TAA

839        // REF=TAAACCCT, ALT=T. T==T => trim. But ALT would become empty (len 0).
840        // No: ALT len is 1, so we stop (keep at least 1 base each).
841        // Actually "T" has len 1 already, so the while loop condition (a.len() > 1) fails.
842        // So after right-trim: REF=TAAACCCTA, ALT=T (trimmed AA from end).
843        // Hmm, let's recount. REF=TAAACCCTAAA (11 chars), ALT=TAA (3 chars).
844        // Right-trim: A==A (both end in A): TAAACCCTAA / TA (10/2)
845        // A==A again: TAAACCCTA / T (9/1). ALT is now len 1, stop.
846        // Left-trim: T==T, but ALT is len 1, can't trim (keep ≥1 each).
847        // Left-align: indel (ALT len 1). Last bases: A vs T, don't match. Done.
848        // Result: pos 105, TAAACCCTA -> T
849        run_test_cases(&[NormTestCase {
850            name: "right_trim_multiallelic_alt1",

862    #[test]
863    fn test_right_trim_multiallelic_alt2() {
864        // 1:105 TAAACCCTAAA -> TAACCCTAAA: right-trim A => TAAACCCTA / TAACCCTA (10/9)
865        // Hmm let me recount. REF=TAAACCCTAAA (11), ALT=TAACCCTAAA (10)
866        // Right-trim: A==A => 10/9. A==A => 9/8. A==A => 8/7. Now:
867        //   REF=TAAACCCT (8), ALT=TAACCCT (7). T==T => 7/6.
868        //   REF=TAAACCC (7), ALT=TAACCC (6). C==C => 6/5.
869        //   REF=TAAACC (6), ALT=TAACC (5). C==C => 5/4.
870        //   REF=TAAAC (5), ALT=TAAC (4). C==C? No: REF ends in C, ALT ends in C. Yes.
871        //   4/3. REF=TAAA (4), ALT=TAA (3). A==A => 3/2. REF=TAA (3), ALT=TA (2).
872        //   A==A => 2/1. REF=TA, ALT=T. ALT len 1, stop.
873        // Left-trim: T==T but ALT is len 1, stop.
874        // Left-align: indel (ALT len 1). Last: A vs T, don't match. Done.
875        // Wait that doesn't match bcftools output. Let me re-read norm.out:
876        // bcftools shows: 1:105 TAAACCCTA -> T,TAACCCTA
877        // So for alt2 (TAACCCTAAA): right-trim AA only (match with REF ending AAA).
878        // Wait, the trimming is per-allele in our implementation.
879        // REF=TAAACCCTAAA (11 bases), ALT=TAACCCTAAA (10 bases)
880        // Right-trim while last bases match:
881        //   both end in A: trim => TAAACCCTAA / TAACCCTAA (10/9)
882        //   both end in A: trim => TAAACCCTA / TAACCCTA (9/8)
883        //   REF ends in A, ALT ends in A: trim => TAAACCCT / TAACCCT (8/7)
884        //   T==T: trim => TAAACCC / TAACCC (7/6)
885        //   C==C: trim => TAAACC / TAACC (6/5)
886        //   C==C: trim => TAAAC / TAAC (5/4)
887        //   C==C: NO! REF[4]=C, ALT[3]=C. Wait: TAAAC ends in C, TAAC ends in C. Yes.
888        //   trim => TAAA / TAA (4/3)
889        //   A==A: trim => TAA / TA (3/2)
890        //   A==A: trim => TA / T (2/1). ALT len 1, stop.
891        //
892        // So right-trim gives: TA / T at pos 105.
893        // Left-trim: T==T, but ALT len 1 => stop.
894        // Left-align: indel. Last bases A vs T, no match. Done.
895        // Result: 105, TA -> T. This is a single-base deletion.
896        //
897        // But bcftools multiallelic output shows TAACCCTA at pos 105 for alt2.
898        // That's because bcftools trims all alleles together to maintain
899        // consistency. Our per-allele approach gives different (but equally
900        // correct) normalization.
901        //
902        // Actually wait — per-allele normalization of this is simply
903        // "delete 1 A from a run of A's in the REF". The result should
904        // left-align to the leftmost A.
905        //
906        // Let's just check: TA -> T at pos 105. Left-align: ALT is "T",
907        // REF is "TA". Last bases: A vs T, don't match. Done.
908        // Result: 105, TA -> T.
909        //
910        // Hmm but actually this should be further left-aligned. The A
911        // being deleted is one of many A's. After right-trimming to TA/T,
912        // we should left-align: last base of REF=A, last base of ALT=T.
913        // They don't match, so no left-alignment. That's correct because
914        // the deletion anchor is T (keeps T, deletes A after it).
915
916        run_test_cases(&[NormTestCase {
917            name: "right_trim_multiallelic_alt2",
918            source:
919                "bcftools:norm.vcf:1:105 TAAACCCTAAA/TAACCCTAAA (MIT, github.com/samtools/bcftools)",
920            chrom: "1",
921            pos: 105,
922            ref_seq: b"TAAACCCTAAA",
923            alt_seq: b"TAACCCTAAA",
924            expected_pos: 105,
925            expected_ref: b"TA",
926            expected_alt: b"T",
927        }]);
